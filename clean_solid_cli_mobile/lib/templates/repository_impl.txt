  @override
  ResultFuture<{{name}}Entity> insert{{name}}({{name}}Entity entity) async {
    return await _executeAction(() => _remoteSource.insert{{name}}(entity));
  }

  ResultFuture<T> _executeAction<T>(Future<T> Function() action) async {
    if (await _networkInfo.isConnected) {
      try {
        final res = await action();
        return Right(res);
      } on ApiException catch (e) {
        return Left(ApiFailure.fromException(e));
      } catch (e) {
        return Left(UnexceptedFailure(e.toString(), "500"));
      }
    }
    return const Left(NetworkFailure("Pas de connexion", "Net_01"));
  }
